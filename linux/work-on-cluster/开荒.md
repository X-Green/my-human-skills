`mkdir ${MYHOME}`

`cd ${MYHOME}`

`mkdir data downl repo tmp tran .local/bin`

# 2. 安装miniconda
``` bash
cd /tmp \
&& curl -O https://repo.anaconda.com/miniconda/Miniconda3-latest-Linux-x86_64.sh \
&& bash ./Miniconda3-latest-Linux-x86_64.sh

source ~/miniconda3/etc/profile.d/conda.sh
conda init zsh
conda init
```

# 3. env: rongyi (不跟/opt/conda混淆)
``` bash
conda create -n rongyi python=3.11
```

# 4. conda->path

``` bash
# .bashrc
export PATH="$HOME/.local/bin:$PATH"
export PATH="$HOME/.conda/envs/rongyi/bin:$PATH"
```
``` bash
# .zshrc
export PATH="$HOME/.local/bin:$PATH"
export PATH="$HOME/.conda/envs/rongyi/bin:$PATH"
```
